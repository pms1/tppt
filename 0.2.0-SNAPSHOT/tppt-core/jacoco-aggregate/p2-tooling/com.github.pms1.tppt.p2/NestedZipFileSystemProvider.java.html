import java.nio.file.attribute.BasicFileAttributes;
import java.nio.file.attribute.FileAttribute;
import java.nio.file.attribute.FileAttributeView;
import java.nio.file.spi.FileSystemProvider;
import java.util.Map;
import java.util.Set;

public class NestedZipFileSystemProvider extends FileSystemProvider {

	@Override
	public String getScheme() {
		return "nested-zip";
	}

	@Override
	public FileSystem newFileSystem(Path path, Map<String, ?> env) throws IOException {
		if (!path.toUri().toString().startsWith("jar:"))
			throw new UnsupportedOperationException();

		Path temp = Files.createTempFile(null, null);

		Files.copy(path, temp, StandardCopyOption.REPLACE_EXISTING);

		FileSystem delegate = FileSystems.newFileSystem(URI.create("jar:" + temp.toUri()), env);

		return new DelegatingFileSystem(delegate) {
			@Override
			public void close() throws IOException {
				try {
					delegate.close();
				} finally {
					Files.delete(temp);
				}
			}
		};
	}
